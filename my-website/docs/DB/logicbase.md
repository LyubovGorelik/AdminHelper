# Схема

```plantuml

@startuml

entity "Клиент" {
    + Номер_телефона : PK
    + Имя
    + email
}

entity "Мастер" {
    + ID : PK
    + Имя
    + Контактная_информация
    + Стаж
}

entity "Заявка" {
    + ID : PK
    + Номер_телефона : FK    + Дата_записи
    + Время_записи
    + Статус
}

entity "Услуга" {
    + ID : PK
    + Название
    + Описание
    + Стоимость
    + Время_выполнения
}

entity "Заявка_Услуга" {
    + ID : PK
    + ID_Заявка : FK
    + ID_Мастер_Услуга : FK
    + Количество
}

entity "Мастер_Услуга" {
    + ID : PK
    + ID_Мастер : FK
    + ID_Услуга : FK
}

entity "Фотоотчет_мастера" {
    + ID : PK
    + ID_Мастер : FK
    + URL_Фотографии
    + Описание
}

entity "Расписание_работы" {
    + ID : PK
    + ID_Мастер : FK
    + Дата
    + Время_начала_работы
    + Время_окончания_работы
}

Клиент ||--o{ Заявка 
Заявка ||--o{ Заявка_Услуга
Мастер_Услуга ||--o{ Заявка_Услуга 
Услуга ||--o{ Мастер_Услуга
Мастер ||--o{ Мастер_Услуга 
Мастер ||--o{ Фотоотчет_мастера
Мастер ||--o{ Расписание_работы

@enduml


```

Логическая модель помогает разработчикам и аналитикам понять требования к данным и спроектировать эффективную и масштабируемую базу данных.
